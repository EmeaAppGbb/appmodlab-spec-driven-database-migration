🐳 GreenHarvest – Docker Compose Infrastructure

docker-compose.yml · SQL Server 2019 source + PostgreSQL 16 target + PostGIS 3.4

📦 docker-compose.yml
version: '3.8'

services:
  sqlserver:
    image: mcr.microsoft.com/mssql/server:2019-latest
    environment:
      ACCEPT_EULA: Y
      SA_PASSWORD: GreenHarvest2024!
      MSSQL_PID:  Developer
    ports:
      - "1433:1433"
    volumes:
      - sqlserver-data:/var/opt/mssql

  postgres:
    image: postgres:16
    environment:
      POSTGRES_DB:       greenharvest
      POSTGRES_USER:     postgres
      POSTGRES_PASSWORD: postgres
    ports:
      - "5432:5432"
    volumes:
      - postgres-data:/var/lib/postgresql/data

  postgis:
    image: postgis/postgis:16-3.4
    environment:
      POSTGRES_DB:       greenharvest
      POSTGRES_USER:     postgres
      POSTGRES_PASSWORD: postgres
    ports:
      - "5433:5432"
    volumes:
      - postgis-data:/var/lib/postgresql/data

volumes:
  sqlserver-data:
  postgres-data:
  postgis-data:
